// Update stock display function updateStockDisplay(productId, newStock) { // Update in product cards const productCards = document.querySelectorAll('.product-card'); productCards.forEach(card => { const addToCartBtn = card.querySelector('button[onclick*="addToCart"]'); if (addToCartBtn && addToCartBtn.getAttribute('onclick').includes(productId)) { const stockSpan = card.querySelector('span[class*="bg-"]'); if (stockSpan) { const stockDisplay = newStock > 10 ? 'In Stock' : `${newStock} remaining`; const stockClass = newStock > 10 ? 'bg-red-100 text-red-800' : newStock > 0 ? 'bg-amber-100 text-amber-800' : 'bg-red-100 text-red-800'; stockSpan.textContent = stockDisplay; stockSpan.className = `px-2 py-1 ${stockClass} rounded-full text-xs font-bold`; if (newStock <= 0) { addToCartBtn.disabled = true; addToCartBtn.classList.add('opacity-50', 'cursor-not-allowed'); addToCartBtn.textContent = 'Out of Stock'; } } } }); }
